Codigo para abrir ventanas
public class VentanaDeAyuda extends javax.swing.JDialog {

    private static VentanaDeAyuda instancia;
    /**
     * Creates new form VentanaAyuda
     */
    private VentanaDeAyuda(java.awt.Frame parent, boolean modal) {
        super(parent, modal);
        initComponents();
    }
    
    public static VentanaDeAyuda obtenerInstancia( java.awt.Frame padre ) {
        if(instancia == null) {
            instancia = new VentanaDeAyuda(padre, false);
        }
        
        instancia.setLocationRelativeTo(padre);
        
        return instancia;
    }

public static void main(String args[]) {
java.awt.EventQueue.invokeLater(new Runnable() {
            public void run() {
                VentanaDeAyuda dialog = new VentanaDeAyuda(new javax.swing.JFrame(), true);
                dialog.addWindowListener(new java.awt.event.WindowAdapter() {
                    @Override
                    public void windowClosing(java.awt.event.WindowEvent e) {
                        System.exit(0);
                    }
                });
                dialog.setVisible(true);
            }
        });
}




private void menuItemTemasDeAyudaActionPerformed(java.awt.event.ActionEvent evt) {                                                     
        VentanaDeAyuda.obtenerInstancia(this).setVisible(true);
    }
